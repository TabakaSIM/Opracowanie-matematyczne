	\Wzor{$z^n=|z|^n(\cos n\varphi + i\sin n\varphi)$}

	$\UkladRownan{\cos \varphi = a}{\sin \varphi = b}$\\
	
	W szczególności zachodzi
	
	\Wzor{$\sqrt[n]{z}=\{\sqrt[n]{z}(\cos j\varphi + i\sin j\varphi) : j\in\{1,2,\dots,n\}\}$}
